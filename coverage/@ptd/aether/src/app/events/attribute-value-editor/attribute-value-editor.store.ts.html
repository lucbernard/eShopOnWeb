import { Injectable, computed, signal } from "@angular/core";
import { AttributeValueType } from "../../system/models/attribute-value-info";
import { AttributeSpecs, AttributeValueMetadata } from "../event-editor/models/attribute-value.model";
 
@Injectable()
export class AttributeValueEditorStore {
  readonly metadata = signal<AttributeValueMetadata | null>(null);
  readonly value = signal<AttributeValueType>(null);
  readonly specs = signal<AttributeSpecs | null>(null);
  readonly enableSpecColorOnValue = signal<boolean>(false);
 
  /**
   * Calculate spec type
   */
  readonly specClass = computed(() => {
    let specType: string | undefined;
 
    if (this.enableSpecColorOnValue()) {
      const value = this.value();
      const specs = this.specs();
      const metadata = this.metadata();
 
      const isNotNull = (value: any) => value !== undefined && value !== null;
      const isNotNullAndLesser = (value: any, spec: any) => isNotNull(value) && isNotNull(spec) && value < spec;
      const isNotNullAndGreater = (value: any, spec: any) => isNotNull(value) && isNotNull(spec) && value > spec;
      const isNotNullAndEqual = (value: any, spec: any) => isNotNull(value) && isNotNull(spec) && value == spec;
 
      const hasValue = isNotNull(value);
 
      const hasUpper =
        specs && (isNotNull(specs.upperUser) || isNotNull(specs.upperWarning) || isNotNull(specs.upperReject) || isNotNull(specs.upperEntry));
 
      const hasTarget = specs && isNotNull(specs.target);
 
      const hasLower =
        specs && (isNotNull(specs.lowerEntry) || isNotNull(specs.lowerReject) || isNotNull(specs.lowerWarning) || isNotNull(specs.lowerUser));
 
      const hasSpecs = hasUpper || hasLower || hasTarget;
 
      if (hasSpecs && hasValue) {
        if (metadata?.inputTypeFriendlyId === "Number") {
          // First, determine if we're outside of a limit.
          if (isNotNullAndLesser(value, specs.lowerEntry)) {
            specType = "lower-entry";
          } else if (isNotNullAndLesser(value, specs.lowerReject)) {
            specType = "lower-reject";
          } else if (isNotNullAndLesser(value, specs.lowerWarning)) {
            specType = "lower-warning";
          } else if (isNotNullAndLesser(value, specs.lowerUser)) {
            specType = "lower-user";
          } else if (isNotNullAndGreater(value, specs.upperEntry)) {
            specType = "upper-entry";
          } else if (isNotNullAndGreater(value, specs.upperReject)) {
            specType = "upper-reject";
          } else if (isNotNullAndGreater(value, specs.upperWarning)) {
            specType = "upper-warning";
          } else if (isNotNullAndGreater(value, specs.upperUser)) {
            specType = "upper-user";
          }
 
          // If we're inside limits.
          if (!specType) {
            if (hasUpper && hasLower) {
              // If we're inside limits and we have a lower and upper limit, then we're in the target range.
              specType = "target";
            } else if (hasUpper && (isNotNullAndGreater(value, specs.target) || isNotNullAndEqual(value, specs.target))) {
              // If we're inside limits and we only have a upper limit, then we're in the target range if we're >= target. < target is undefined (no color).
              specType = "target";
            } else if (hasLower && (isNotNullAndLesser(value, specs.target) || isNotNullAndEqual(value, specs.target))) {
              // If we're inside limits and we only have a lower limit, then we're in the target range if we're <= target. > target is undefined (no color).
              specType = "target";
            } else if (!hasLower && !hasUpper && isNotNullAndEqual(value, specs.target)) {
              // If we do not have a lower limit nor upper limit, then we're in the target if we're on target (ex: setpoint).
              specType = "target";
            }
          }
        }
 
        if (
          metadata?.inputTypeFriendlyId === "Text" ||
          metadata?.inputTypeFriendlyId === "DropDown" ||
          metadata?.inputTypeFriendlyId === "ComboBox"
        ) {
          // Only color-code if exact match to limit
          if (isNotNullAndEqual(value, specs.lowerEntry)) {
            specType = "lower-entry";
          } else if (isNotNullAndEqual(value, specs.lowerReject)) {
            specType = "lower-reject";
          } else if (isNotNullAndEqual(value, specs.lowerWarning)) {
            specType = "lower-warning";
          } else if (isNotNullAndEqual(value, specs.lowerUser)) {
            specType = "lower-user";
          } else if (isNotNullAndEqual(value, specs.target)) {
            specType = "target";
          } else if (isNotNullAndEqual(value, specs.upperUser)) {
            specType = "upper-user";
          } else if (isNotNullAndEqual(value, specs.upperWarning)) {
            specType = "upper-warning";
          } else if (isNotNullAndEqual(value, specs.upperReject)) {
            specType = "upper-reject";
          } else if (isNotNullAndEqual(value, specs.upperEntry)) {
            specType = "upper-entry";
          }
        }
      }
    }
 
    return specType;
  });
}
