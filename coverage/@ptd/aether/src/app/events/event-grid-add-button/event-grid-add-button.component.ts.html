  private readonly _assets = inject(AssetService);
  private readonly _auth = inject(AuthService);
 
  readonly configuration = input.required<EventGridAddOptions, EventGridAddOptions>({ transform: EventGridAddOptionsDefault });
 
  readonly size = computed(() => this.configuration().size);
 
  readonly options = signal<Array<EventGridAddOption>>([]); // Options are computed from the configuration and the access type, to filter out the ones that the user does not have the right to
  readonly isSingleOption = computed(() => this.options().length === 1);
  readonly visible = computed(() => this.options().length > 0);
  readonly defaultCaption = signal<Localize>({ key: "common.action.add" });
  readonly caption = computed(() => (this.options().length === 1 ? this.options()[0].caption : this.defaultCaption()));
 
  readonly saving = input<boolean>(false);
  readonly disabled = computed(() => this.saving());
 
  @Output()
  readonly requestAdd = new EventEmitter<EventGridAddButtonRequestAdd>();
 
  /**
   * Constructor
   */
  constructor() {
    combineLatest([this._auth.accessTypeReadWriteLevel$, toObservable(this.configuration)])
      .pipe(
        switchMap(([accessTypeReadWriteLevel, configuration]) => {
          const assetIds = (configuration.options.map((m) => m.assetId) || []).filter(
            (item): item is string | number => item !== null && item !== undefined
          );
 
          return (assetIds.length ? this._assets.getAssetAccessType(assetIds) : of([])).pipe(
            tap((accessTypes) => {
              this.options.set(
                configuration.options.filter(
                  (option) =>
                    // Only keep the options where there's no asset and the configuration is a full "add"
                    // Those means we don't know the asset and we can't pre-check the security
                    ((option.assetId === undefined || option.assetId === null) &&
                      (option.creationMode === "add" || option.creationMode === "addAndEdit")) ||
                    // Or for the other options, we keep the ones where the user has the right to add
                    accessTypes.some(
                      (s) =>
                        (option.assetId === s.assetId || option.assetId === s.assetFriendlyId) &&
                        (s.level || Number.MIN_SAFE_INTEGER) >= accessTypeReadWriteLevel
                    )
                ) || []
              );
            })
          );
        }),
        takeUntilDestroyed()
      )
      .subscribe();
  }
