// Possible values for the command type, note that these should all be lowercase with current implementation
const eventColumnCommandTypes = ["open", "get", "edit", "delete", "splitdowntime", "comment"] as const;
export type EventColumnCommandType = (typeof eventColumnCommandTypes)[number];
 
type PrefixWith<T extends string, Prefix extends string> = `${Prefix}.${T}`;
// Possible values for the command type in all scopes
export type EventColumnScopedCommandType =
  | EventColumnCommandType
  | PrefixWith<EventColumnCommandType, "parent">
  | PrefixWith<EventColumnCommandType, "child">;
 
/**
 * Command config
 * Commands are shown under a `...` dropdown button in the column and emit an event when clicked
 * Caption is optional and will be the text of the command in the dropdown. A default caption is used if not provided
 *
 * @examples [{"command":"edit","caption":"Edit"},{"command":"delete","caption":"Delete"}]
 */
export interface EventCommandConfig {
  command: EventColumnScopedCommandType;
  caption?: Localize;
}
 
/**
 * Configuration for commands of an event
 * Commands are shown under a `...` dropdown button in the column and emit an event when clicked
 * Handler of the event is the integrator
 *
 * @examples [{"commands":["edit","delete",`splitDowntime`]},{"commands":["edit"]}]
 */
export interface EventCommandColumnConfiguration {
  columnType: "command"; // Override the column type to be command
 
  /**
   * The command is an array of string. Can be one or all of `open`, `edit`, `get`, `delete`, `comment`, `splitDowntime`.
   * It can also be an array of object with a `command` and `caption` property. Caption is optional and will be the text of the command in the dropdown. A default caption is used if not provided
   * It can also be prefixed with 'parent' or 'child' to show the value of the parent or child event of the row's main event. Ex: 'parent.events.startTime'. This will force event grid into genealogy mode.
   *
   * Default to []
   *
   * @examples [["edit","delete","splitDowntime","child.open", {"command": "parent.edit", "caption": "Edit Parent"}]]
   */
  commands: Array<EventColumnScopedCommandType | EventCommandConfig>;
}
 
/**
 * Configuration of a property of an event
 *
 * @examples [{"property":"events.startTime","title":"Start Time","type":"date","format":"medium"},{"property":"events.endTime","title":"End Time","type":"date","format":"medium"},{"property":"assets.displayName","title":"Asset Name","type":"text"},{"property":"attributeValues.Temperature","title":"Temperature","type":"numeric","format":"1.2-2]
 */
export type EventPropertyColumnConfiguration = {
  columnType: "property"; // Override the column type to be property
 
  /**
   * The Property is the data of column. Must be composed of the combination `dbo.table` + `column` of that table separated by ".", like `events.StartTime`.
   * To show the value of an attributes on a event, the table prefix to use is `attributesValues` and the column is the friendly id of the attributes, like `attributeValues.Temperature`
   * In the case you want to display the columns that must be composed of two or more tables,
   * It can also be prefixed with 'parent' or 'child' to show the value of the parent or child event of the row's main event. Ex: 'parent.events.startTime'. This will force event grid into genealogy mode.
   *
   * Required.
   *
   * @examples ["events.startTime","events.endTime","assets.displayName","attributeValues.Temperature"]
   */
  property: string;
  /**
   * Title of the column.
   * If parameter is a string, the value is kept as is
   * If parameter is a translation key, the value is translated on the client side using the client translations
   * If parameter is a database localization key, the value will be translated by the messages from the database
   *
   * @examples ["Start Time", "Event Name", "Asset Name", {"key": "model.events.properties.startTime.property"}, { "module": "Aether_CoreUX_WO", "friendlyId": "MaskCompletedWO" }]
   */
  title?: Localize;
  /**
   * Type of data
   * Default to text
   * For event-list, type are automatically handled by their type in the database and are optional
   *
   * @examples ["text","numeric","boolean","date"]
   */
  type: GridPropertyColumnType;
  /**
   * Enables an user to edit the value. Only supported for attributes
   * Optional. Default to false.
   *
   * @examples [true, false]
   */
  editable: boolean;
  /**
   * Formatting of the value. Supported only on numeric and date column
   *
   * For `date`, supported format are: https://angular.io/api/common/DatePipe#usage-notes
   * For `numeric`, supported format are: https://angular.io/api/common/DecimalPipe#usage-notes
   *
   * For `date`, the default format is `medium`
   */
  format?: string;
  /**
   * Internal component used. Not supported for integrator
   */
  component?: string;
  /**
   * Configuration of the component.
   * Format of the field is specific to the component / special field used.
   */
  configuration?: any;
};
