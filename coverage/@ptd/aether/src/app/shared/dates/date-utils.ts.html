/**
 * A function that adds and subtracts seconds from a `Date` object.
 *
 * @param date - The initial date value.
 * @param offset - The number of seconds to add and subtract from the date.
 * @returns - A new `Date` instance.
 *
 * @example
 * ```ts-no-run
 * addSeconds(new Date(2016, 0, 1), 5); // 2016-1-1 00:00:05
 * addSeconds(new Date(2016, 0, 1), -5); // 2015-12-31 23:59:55
 * ```
 */
export function addSeconds(date: Date, offset: number) {
  return new Date(date.getTime() + offset * SECONDS_MS);
}
